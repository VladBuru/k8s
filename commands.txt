kubectl create -f pod.yaml
    команда на создание пода в кластере кубера
    вызвав больше одного раза эту команду на манифест с одинаковым name - будет error и под не создастся
    kubectl - тулза (инструмент)
    create - создать
    -f - из файловой системы
    pod.yaml - имя файла манифеста

kubectl apply -f pod.yaml
    команда на создание либо обновление пода (как CREATE or UPDATE).
    Если мы поменяли манифест, то использовать эту команду а не create.

kubectl get pod
kubectl get po
    команда показывает поднятые поды
kubectl get replicaset
kubectl get rs
    команда показывает поднятые репликасеты
kubectl get deployment
kubectl get deploy
    команда показывает поднятые deployment
kubectl get service
kubectl get svc
    команда показывает поднятые Service
kubectl get endpoint //ep
kubectl get po -o wide
    показывает расширенную информацию о подах

kubectl delete pod --all
kubectl delete pod my-pod-q73fk
    --all - удалить все поды (лучше не использовать на проде)
    my-pod-q73fk - удалить по имени пода

kubectl scale --replicas=2 replicaset my-replicaset
    Команда на изменение какого либо параметра в манифесте.
    Не меняет сам манифест, а изменяет например количество реплик в рантайме через командную строку.
    Лучше так не делать, так как реальное количество поднятых реплик будет не стыковаться с манифестом
    Могут использовать девопсы чтобы в рантайме что то поменять временно

kubectl describe replicaset my-replicaset
kubectl describe configmap my-configmap
    Команда показывает всю информацию о репликасете или другом объекте k8s

kubectl rollout undo deployment my-deployment
    Откатывает версию манифеста deployment к предыдущей. Например меняем версию - делаем apply и потом эту команду.
    Это можно сделать за счет того, что Deployment хранит старые версии ReplicaSet,
    чтобы можно было к ним вернуться

kubectl apply -f configmap.yaml -f deployment-configmap.yaml
    поднимается конфиг мапа и deployment

kubectl exec -it my-deployment-7fdb7c5f85-j6f4n -- bash
    Команда, чтобы зайти внутрь пода.
    Можно посмотреть структуру файловой системы, конфиги, ввести команды изнутри контейнера и т.д.

kubectl port-forward my-deployment-7fdb7c5f85-j6f4n 8005:80
    Прокидываем порты. Внешний 8085 сможет связатсья с внутренним портом контейнера 80

kubectl edit cm my-configmap
    Редактирование конфиг мапы. cm - configmap

kubectl rollout restart deployment my-deployment
    Рестарт deployment

kubectl run -t -i --rm --image amouat/network-utils test -- bash
    Создает временный под с интерактивным bash для тестирования связи с другими подами и сетевых утилит
    curl my-service - стучимся к my-service. Можно много раз - будет видно
    что на запрос отвечает не один и тот же порт, а несколько (балансировка нагрузки между репликами)

kubectl apply -f https://raw.githubusercontent.com/kubernetes/ingress-nginx/main/deploy/static/provider/cloud/deploy.yaml
    подтягиваем и устанавливаем Ingress-контроллер

kubectl describe vpa my-app-vpa
    показывает детальную информацию о VPA а также рекомендует оптимальные настройки min and max ресурсов

kubectl get pods -A --kubeconfig=config_vburula
    показать все поды по определенному конфигу (в конфиге указывается неймспейс, данные для подключения к определенному кластеру, стенду

kubectl create namespace vburula --dry-run=server --kubeconfig=config_vburula
    моковое создание неймспейса (проверяем в логах что успешно создали, но по факту не создается), если все норм - создаем
kubectl create namespace vburula --kubeconfig=config_vburula
    реальная команда на создание
kubectl create namespace vburula --dry-run=server -o yaml --kubeconfig=config_vburula
    команда еще предоставит ответ -o (output) как будет выглядеть манифест

kubectl create -f pod.yaml --kubeconfig=config_vburula
kubectl get pod --kubeconfig=config_vburula
kubectl delete pod my-pod --kubeconfig=config_vburula
kubectl create -f replicaset.yaml --kubeconfig=config_vburula
и так далее с параметром --kubeconfig=config_vburula вызываем команды чтобы работать с удаленным кубером
